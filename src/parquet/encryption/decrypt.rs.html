33pub fn read_and_decrypt<T: Read>(
34    decryptor: &Arc<dyn BlockDecryptor>,
35    input: &mut T,
36    aad: &[u8],
37) -> Result<Vec<u8>> {
38    let mut len_bytes = [0; 4];
39    input.read_exact(&mut len_bytes)?;
40    let ciphertext_len = u32::from_le_bytes(len_bytes) as usize;
41    let mut ciphertext = vec![0; 4 + ciphertext_len];
42    input.read_exact(&mut ciphertext[4..])?;
43
44    decryptor.decrypt(&ciphertext, aad.as_ref())
45}

47// CryptoContext is a data structure that holds the context required to
48// decrypt parquet modules (data pages, dictionary pages, etc.).
49#[derive(Debug, Clone)]
50pub(crate) struct CryptoContext {
51    pub(crate) row_group_idx: usize,
52    pub(crate) column_ordinal: usize,
53    pub(crate) page_ordinal: Option<usize>,
54    pub(crate) dictionary_page: bool,
55    // We have separate data and metadata decryptors because
56    // in GCM CTR mode, the metadata and data pages use
57    // different algorithms.
58    data_decryptor: Arc<dyn BlockDecryptor>,
59    metadata_decryptor: Arc<dyn BlockDecryptor>,
60    file_aad: Vec<u8>,
61}
62
63impl CryptoContext {
64    pub(crate) fn for_column(
65        file_decryptor: &FileDecryptor,
66        column_crypto_metadata: &ColumnCryptoMetaData,
67        row_group_idx: usize,
68        column_ordinal: usize,
69    ) -> Result<Self> {
70        let (data_decryptor, metadata_decryptor) = match column_crypto_metadata {
71            ColumnCryptoMetaData::EncryptionWithFooterKey => {
72                // TODO: In GCM-CTR mode will this need to be a non-GCM decryptor?
73                let data_decryptor = file_decryptor.get_footer_decryptor()?;
74                let metadata_decryptor = file_decryptor.get_footer_decryptor()?;
75                (data_decryptor, metadata_decryptor)
76            }
77            ColumnCryptoMetaData::EncryptionWithColumnKey(column_key_encryption) => {
78                let key_metadata = &column_key_encryption.key_metadata;
79                let full_column_name;
80                let column_name = if column_key_encryption.path_in_schema.len() == 1 {
81                    &column_key_encryption.path_in_schema[0]
82                } else {
83                    full_column_name = column_key_encryption.path_in_schema.join(".");
84                    &full_column_name
85                };
86                let data_decryptor = file_decryptor
87                    .get_column_data_decryptor(column_name, key_metadata.as_deref())?;
88                let metadata_decryptor = file_decryptor
89                    .get_column_metadata_decryptor(column_name, key_metadata.as_deref())?;
90                (data_decryptor, metadata_decryptor)
91            }
92        };
93
94        Ok(CryptoContext {
95            row_group_idx,
96            column_ordinal,
97            page_ordinal: None,
98            dictionary_page: false,
99            data_decryptor,
100            metadata_decryptor,
101            file_aad: file_decryptor.file_aad().clone(),
102        })
103    }
104
105    pub(crate) fn with_page_ordinal(&self, page_ordinal: usize) -> Self {
106        Self {
107            row_group_idx: self.row_group_idx,
108            column_ordinal: self.column_ordinal,
109            page_ordinal: Some(page_ordinal),
110            dictionary_page: false,
111            data_decryptor: self.data_decryptor.clone(),
112            metadata_decryptor: self.metadata_decryptor.clone(),
113            file_aad: self.file_aad.clone(),
114        }
115    }
116

118        let module_type = if self.dictionary_page {
119            ModuleType::DictionaryPageHeader
120        } else {
121            ModuleType::DataPageHeader
122        };
123
124        create_module_aad(
125            self.file_aad(),
126            module_type,
127            self.row_group_idx,
128            self.column_ordinal,
129            self.page_ordinal,
130        )
131    }
132
133    pub(crate) fn create_page_aad(&self) -> Result<Vec<u8>> {
134        let module_type = if self.dictionary_page {
135            ModuleType::DictionaryPage
136        } else {
137            ModuleType::DataPage
138        };
139
140        create_module_aad(
141            self.file_aad(),
142            module_type,
143            self.row_group_idx,
144            self.column_ordinal,
145            self.page_ordinal,
146        )
147    }
148
149    pub(crate) fn for_dictionary_page(&self) -> Self {
150        Self {
151            row_group_idx: self.row_group_idx,
152            column_ordinal: self.column_ordinal,
153            page_ordinal: self.page_ordinal,
154            dictionary_page: true,
155            data_decryptor: self.data_decryptor.clone(),
156            metadata_decryptor: self.metadata_decryptor.clone(),
157            file_aad: self.file_aad.clone(),
158        }
159    }
160
161    pub(crate) fn data_decryptor(&self) -> &Arc<dyn BlockDecryptor> {
162        &self.data_decryptor
163    }
164
165    pub(crate) fn file_aad(&self) -> &Vec<u8> {
166        &self.file_aad
167    }

170#[derive(Clone, PartialEq)]
171struct ExplicitDecryptionKeys {
172    footer_key: Vec<u8>,
173    column_keys: HashMap<String, Vec<u8>>,
174}
175
176#[derive(Clone)]
177enum DecryptionKeys {
178    Explicit(ExplicitDecryptionKeys),
179    ViaRetriever(Arc<dyn KeyRetriever>),
180}
181
182impl PartialEq for DecryptionKeys {
183    fn eq(&self, other: &Self) -> bool {
184        match (self, other) {
185            (DecryptionKeys::Explicit(keys), DecryptionKeys::Explicit(other_keys)) => {
186                keys.footer_key == other_keys.footer_key
187                    && keys.column_keys == other_keys.column_keys
188            }
189            (DecryptionKeys::ViaRetriever(_), DecryptionKeys::ViaRetriever(_)) => true,
190            _ => false,
191        }
192    }
193}

195/// FileDecryptionProperties hold keys and AAD data required to decrypt a Parquet file.
196#[derive(Clone, PartialEq)]
197pub struct FileDecryptionProperties {
198    keys: DecryptionKeys,
199    pub(crate) aad_prefix: Option<Vec<u8>>,
200}
201
202impl FileDecryptionProperties {
203    /// Returns a new [`FileDecryptionProperties`] builder that will use the provided key to
204    /// decrypt footer metadata.
205    pub fn builder(footer_key: Vec<u8>) -> DecryptionPropertiesBuilder {
206        DecryptionPropertiesBuilder::new(footer_key)
207    }
208
209    /// Returns a new [`FileDecryptionProperties`] builder that uses a [`KeyRetriever`]
210    /// to get decryption keys based on key metadata.
211    pub fn with_key_retriever(key_retriever: Arc<dyn KeyRetriever>) -> DecryptionPropertiesBuilder {
212        DecryptionPropertiesBuilder::new_with_key_retriever(key_retriever)
213    }
214
215    /// Get the encryption key for decrypting a file's footer,
216    /// and also column data if uniform encryption is used.
217    pub(crate) fn footer_key(&self, key_metadata: Option<&[u8]>) -> Result<Cow<Vec<u8>>> {
218        match &self.keys {
219            DecryptionKeys::Explicit(keys) => Ok(Cow::Borrowed(&keys.footer_key)),
220            DecryptionKeys::ViaRetriever(retriever) => {
221                let key = retriever.retrieve_key(key_metadata.unwrap_or_default())?;
222                Ok(Cow::Owned(key))
223            }
224        }
225    }
226
227    /// Get the column-specific encryption key for decrypting column data and metadata within a file
228    pub(crate) fn column_key(
229        &self,
230        column_name: &str,
231        key_metadata: Option<&[u8]>,
232    ) -> Result<Cow<Vec<u8>>> {
233        match &self.keys {
234            DecryptionKeys::Explicit(keys) => match keys.column_keys.get(column_name) {
235                None => Err(general_err!(
236                    "No column decryption key set for column '{}'",
237                    column_name
238                )),
239                Some(key) => Ok(Cow::Borrowed(key)),
240            },
241            DecryptionKeys::ViaRetriever(retriever) => {
242                let key = retriever.retrieve_key(key_metadata.unwrap_or_default())?;
243                Ok(Cow::Owned(key))
244            }
245        }
246    }
247}
248
249impl std::fmt::Debug for FileDecryptionProperties {
250    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
251        write!(f, "FileDecryptionProperties {{ }}")
252    }
253}

255/// Builder for [`FileDecryptionProperties`]
256pub struct DecryptionPropertiesBuilder {
257    footer_key: Option<Vec<u8>>,
258    key_retriever: Option<Arc<dyn KeyRetriever>>,
259    column_keys: HashMap<String, Vec<u8>>,
260    aad_prefix: Option<Vec<u8>>,
261}
262
263impl DecryptionPropertiesBuilder {
264    /// Create a new [`DecryptionPropertiesBuilder`] builder that will use the provided key to
265    /// decrypt footer metadata.
266    pub fn new(footer_key: Vec<u8>) -> DecryptionPropertiesBuilder {
267        Self {
268            footer_key: Some(footer_key),
269            key_retriever: None,
270            column_keys: HashMap::default(),
271            aad_prefix: None,
272        }
273    }
274
275    /// Create a new [`DecryptionPropertiesBuilder`] by providing a [`KeyRetriever`] that
276    /// can be used to get decryption keys based on key metadata.
277    pub fn new_with_key_retriever(
278        key_retriever: Arc<dyn KeyRetriever>,
279    ) -> DecryptionPropertiesBuilder {
280        Self {
281            footer_key: None,
282            key_retriever: Some(key_retriever),
283            column_keys: HashMap::default(),
284            aad_prefix: None,
285        }
286    }
287
288    /// Finalize the builder and return created [`FileDecryptionProperties`]
289    pub fn build(self) -> Result<FileDecryptionProperties> {
290        let keys = match (self.footer_key, self.key_retriever) {
291            (Some(footer_key), None) => DecryptionKeys::Explicit(ExplicitDecryptionKeys {
292                footer_key,
293                column_keys: self.column_keys,
294            }),
295            (None, Some(key_retriever)) => {
296                if !self.column_keys.is_empty() {
297                    return Err(general_err!(
298                        "Cannot specify column keys directly when using a key retriever"
299                    ));
300                }
301                DecryptionKeys::ViaRetriever(key_retriever)
302            }
303            _ => {
304                unreachable!()
305            }
306        };
307        Ok(FileDecryptionProperties {
308            keys,
309            aad_prefix: self.aad_prefix,
310        })
311    }
312
313    /// Specify the expected AAD prefix to be used for decryption.
314    /// This must be set if the file was written with an AAD prefix and the
315    /// prefix is not stored in the file metadata.
316    pub fn with_aad_prefix(mut self, value: Vec<u8>) -> Self {
317        self.aad_prefix = Some(value);
318        self
319    }
320
321    /// Specify the decryption key to use for a column
322    pub fn with_column_key(mut self, column_name: &str, decryption_key: Vec<u8>) -> Self {
323        self.column_keys
324            .insert(column_name.to_string(), decryption_key);
325        self
326    }
327}

329#[derive(Clone, Debug)]
330pub(crate) struct FileDecryptor {
331    decryption_properties: FileDecryptionProperties,
332    footer_decryptor: Arc<dyn BlockDecryptor>,
333    file_aad: Vec<u8>,
334}
335
336impl PartialEq for FileDecryptor {
337    fn eq(&self, other: &Self) -> bool {
338        self.decryption_properties == other.decryption_properties && self.file_aad == other.file_aad
339    }
340}
341
342impl FileDecryptor {
343    pub(crate) fn new(
344        decryption_properties: &FileDecryptionProperties,
345        footer_key_metadata: Option<&[u8]>,
346        aad_file_unique: Vec<u8>,
347        aad_prefix: Vec<u8>,
348    ) -> Result<Self> {
349        let file_aad = [aad_prefix.as_slice(), aad_file_unique.as_slice()].concat();
350        let footer_key = decryption_properties.footer_key(footer_key_metadata)?;
351        let footer_decryptor = RingGcmBlockDecryptor::new(&footer_key).map_err(|e| {
352            general_err!(
353                "Invalid footer key. {}",
354                e.to_string().replace("Parquet error: ", "")
355            )
356        })?;
357
358        Ok(Self {
359            footer_decryptor: Arc::new(footer_decryptor),
360            decryption_properties: decryption_properties.clone(),
361            file_aad,
362        })
363    }
364
365    pub(crate) fn get_footer_decryptor(&self) -> Result<Arc<dyn BlockDecryptor>> {
366        Ok(self.footer_decryptor.clone())
367    }
368
369    pub(crate) fn get_column_data_decryptor(
370        &self,
371        column_name: &str,
372        key_metadata: Option<&[u8]>,
373    ) -> Result<Arc<dyn BlockDecryptor>> {
374        let column_key = self
375            .decryption_properties
376            .column_key(column_name, key_metadata)?;
377        Ok(Arc::new(RingGcmBlockDecryptor::new(&column_key)?))
378    }
379
380    pub(crate) fn get_column_metadata_decryptor(
381        &self,
382        column_name: &str,
383        key_metadata: Option<&[u8]>,
384    ) -> Result<Arc<dyn BlockDecryptor>> {
385        // Once GCM CTR mode is implemented, data and metadata decryptors may be different
386        self.get_column_data_decryptor(column_name, key_metadata)
387    }
388
389    pub(crate) fn file_aad(&self) -> &Vec<u8> {
390        &self.file_aad
391    }
